CFR#659 (Div 2)

A. Common Prefixes

题意

原题：CF1384 div2 A 【Common Prefixes】

  • 按顺序给定 nnn 个正整数 a1...ana_1 ... a_na1​...an​，为相邻两个字符串的最长公共前缀长度。
  • 求这 n+1n+1n+1 个字符串。

分析

我们先用 aaa 到 zzz 这所有字母顺次填充长度为 200200200 的初始串。

对于每 aia_iai​ ，把串的第 ai+1a_i + 1ai​+1 位修改一下，比如改成它的下一个字母 (z 变成 a) 。

然后就做完了。

源码

#include <bits/stdc++.h>

using namespace std;

const int N = 105;

int n;
char s[2*N];

int main()
{
	int T;
	scanf("%d", &T);
	while (T--) {
		for (int i = 1; i <= 200; i++)
			s[i] = (i%26?i%26:26) + 'a' - 1;
		
		scanf("%d", &n);
		for (int i = 1; i <= 200; i++)
			printf("%c", s[i]);
		printf("\n");
		
		for (int i = 1; i <= n; i++) {
			int x;
			scanf("%d", &x);
			x++;
			s[x] = (s[x]+1) > 'z' ? 'a' : s[x]+1;
		
			for (int j = 1; j <= 200; j++) {
				printf("%c", s[j]);
			}
			
			printf("\n");
		}
	}
	
	return 0;
}

B. Koa and the Beach

题意

原题：CF1384 div2 B 【Koa and the Beach】

B1

数据范围小，可以直接 DP。

具体地，设 f[i][j]f[i][j]f[i][j] 表示 jjj 时刻在 iii 处是否可行，其中 jjj 为时间对 2k2k2k 取模的结果。

然后按题意递推即可。注意可以停留在原地，意味着我们可能可以通过 f[i][j]f[i][j]f[i][j] 更新 f[i][k]    (k>j)f[i][k] \;\;(k > j)f[i][k](k>j) 的值。具体见代码。

时间复杂度 O(nk)O(nk)O(nk) 。

#include <bits/stdc++.h>

using namespace std;

const int N = 105, K = 205, inf = 2e9;

int n, k, m;
int a[N];
int f[N][K];

int main()
{
	int T;
	scanf("%d", &T);
	while (T--) {
		scanf("%d%d%d", &n, &k, &m);
		for (int i = 1; i <= n; i++) {
			scanf("%d", &a[i]);
		}
		
		memset(f, 0, sizeof(f));
		for (int i = 0; i <= 2*k-1; i++)
			f[0][i] = 1;
		a[n+1] = -inf;
			
		for (int i = 0; i <= n; i++) {
			for (int j = 0; j <= 2*k-1; j++) {
				if (f[i][j]) {
					int t = (j+1) % (2*k);
					if (a[i] + (t <= k ? t : 2*k - t) <= m) f[i][t] = 1;
				}
			}
			
			for (int j = 0; j <= 2*k-1; j++) {
				int t = (j+1) % (2*k);
//				printf("i = %d, j = %d, d = %d\n", i+1, t, a[i+1] + (t<=k?t:2*k-t) );
				if (a[i+1] + (t<=k?t:2*k-t) <= m) {
					f[i+1][t] |= f[i][j];
				}
			}
//			for (int j = 0; j <= 2*k-1; j++) {
//				printf("f[%d][%d] = %d ", i+1, j, f[i+1][j]);
//			}printf("\n");
		}
		
		int flag = 0;
		for (int i = 0; i <= 2*k-1; i++) {
			if (f[n+1][i]) {
				flag = 1;
				break;
			}
		}
		
		if (flag) {
			puts("Yes");
		}
		else {
			puts("No");
		}
	}
	
	return 0;
}

B2

数据范围较大，不能用上述 DP 。

记 t[i]=min⁡(l−a[i],k)t[i] = \min(l - a[i], k)t[i]=min(l−a[i],k) ，那么对于每个点，最高可以接受 t[i]t[i]t[i] 高的潮水。用时间描述，就是：

  • 若 t[i]=kt[i] = kt[i]=k ，意味着可以在任意时刻处于 iii 号点；
  • 若 0≤t[i]<k0 \le t[i] < k0≤t[i]<k ，意味着可以在 [0,t[i]]\Big[0, t[i]\Big][0,t[i]] 和 [2k−t[i],2k−1]\Big[2k-t[i], 2k-1\Big][2k−t[i],2k−1] 的时间段处于 iii 号点；
  • 若 t[i]<0t[i] < 0t[i]<0 则不能处于 iii 号点。

注意，上述的时间均是在模 2k2k2k 意义下的。

可以考虑贪心。

我们用一个 nownownow 表示当前处在哪一时刻。如果能够抵达当前点（t[i]<0t[i] < 0t[i]<0 直接判不行并退出），

  • t[i]=kt[i] = kt[i]=k ，根据贪心策略，我们直接令 now=2k−t[i+1]now = 2k-t[i+1]now=2k−t[i+1] 是最优的，并且进入下一个点的讨论；
  • 0≤t[i]<k0 \le t[i] < k0≤t[i]<k ，同样，我们应该在满足上一轮操作的条件下，尽可能把 nownownow 放在退潮时间段的靠前位置，或者涨潮时间段的靠前位置，才能为后面争取更优的状态。因此，讨论一下 nownownow 是在涨潮还是退潮时间段，进而得出其取值。具体见代码。
  • 在第二种情况中，如果选出的最优的 nownownow 都不满足当前点的可行时间段，就判不行并退出。

时间复杂度 O(n)O(n)O(n)

当时思路不是很清晰，打出来的代码有点丑与凌乱。

#include <bits/stdc++.h>

using namespace std;

const int N = 3e5+5;

int n, m, k;
int a[N];
int t[N];

int main()
{
	int T;
	scanf("%d", &T);
	while (T--) {
		scanf("%d%d%d", &n, &k, &m);
		for (int i = 1; i <= n; i++) {
			scanf("%d", &a[i]);
			a[i] = m - a[i];
			
			if (a[i] < 0) {
				t[i] = -1;
			}
			else {
				t[i] = min(k, a[i]);
			}
		}
		
		int flag, now;
		for (int i = 1; i <= n; i++) {
			if (i == 1) {
				if (t[i] >= 0)
					flag = 1;
				else {
					flag = 0;
					break;
				}
				
				if (t[i] == k)
					now = -1;
				else
					now = 2*k - t[i];
				continue;
			}
		
			if (now == -1)
				now = 2*k - t[i];
			else {
				if (now < k) 
					now = now+1;
				else {
					now = max(now+1, 2*k - t[i]);
				}
			}
				
			now %= 2*k;
			
			if (t[i] < 0) {
				flag = 0;
				break;
			}
			
			if (t[i] == k) {
				flag = 1;
				now = -1;
			}
			else {
				if (now <= t[i] || now >= 2*k-t[i]) {
					flag = 1;
				}
				else {
					flag = 0;
					break;
				}
			}
			
		}
		
		if (flag) {
			puts("Yes");
		}
		else {
			puts("No");
		}
		
	}
	
	return 0;
}

C. String Transformation 1

题意

原题：CF1384 div2 C 【String Transformation 1】

  • 给定长度为 nnn 的两字符串 AAA ，BBB ，由 a...ta...ta...t 这 202020 个小写字母组成。
  • 每次选择 AAA 中若干个相同的字母 xxx 一起变成字母 yyy ，要求 (y>x)(y > x)(y>x) 并且 yyy 在 a...ta...ta...t 之间。这称为一次操作。
  • 求最少操作数使 AAA 变成 BBB 。
  • 多组测试

分析

由于每次只能把字符变大，所以容易发现有解的充要条件是 AAA 中的字母 比 在其 BBB 中的对应位置的字母 小。

贪心策略：每次选择 AAA 中的所有相同字母 Ap1,Ap2,...,ApkA_{p_1}, A_{p_2}, ... , A_{p_k}Ap1​​,Ap2​​,...,Apk​​ ，一起变成 min⁡(Bp1,Bp2,...BPk)\min\Big(B_{p_1}, B_{p_2}, ... B_{P_k}\Big)min(Bp1​​,Bp2​​,...BPk​​) 。

这样，可以充分利用每次操作，并且不会比其它方式更劣。

所以分别枚举 s, ts,\, ts,t，如果满足上述贪心策略就把 AAA 中的 sss 全部变成 ttt 。
时间复杂度 O(w2n)O(w^2n)O(w2n)，其中 www 为字符值域大小，即 202020 。

源码

#include <bits/stdc++.h>

using namespace std;

const int N = 1e5+5;

int n;
int a[N], b[N];
int ans = 0;
int cnt[30][30];
char s[N];

int main()
{
	int T;
	scanf("%d", &T);
	while (T--) {
		ans = 0;
		memset(cnt, 0, sizeof(cnt));
		
		scanf("%d", &n);
		scanf("%s", s+1);
		for (int i = 1; i <= n; i++)
			a[i] = s[i] - 'a' + 1;
		scanf("%s", s+1);
		for (int i = 1; i <= n; i++) 
			b[i] = s[i] - 'a' + 1;
		
		int flag = 1;
		for (int i = 1; i <= n; i++) {
			if (a[i] > b[i]) {
				flag = 0;
				break;
			}
		}
		
		if (!flag) {
			printf("-1\n");
			continue;
		}
		
		for (int t = 1; t <= 20; t++) {
			for (int r = t+1; r <= 20; r++) {
				int f = 0;
				for (int i = 1; i <= n; i++) {
					if (a[i] == t && b[i] == r) {
						f = 1;
						break;
					}
				}
				
				if (!f) continue;
				
				for (int i = 1; i <= n; i++) {
					if (a[i] == t && b[i] >= r) {
						a[i] = r;
					}
				}
				ans++;
			}
		}
		
		printf("%d\n", ans);
		
	} 
	return 0;
}
