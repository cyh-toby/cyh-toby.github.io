A 儒略日

题意

原题：[CSP-S2020] 儒略日

  • 给定日期计算规则和起始日期（公元前 4713 年 1 月 1 日）和天数，求日期

  • 多组询问

分析

可以考虑把询问离线。

在复杂的那一段时间（ 2000 年 1 月 1 日以前），直接枚举年份暴力模拟，并回答相应询问。

之后暴力模拟 400 年，找到天数周期，然后 O(1)O(1)O(1) 回答剩下的询问。

源码

比较长。

const int Q = 1e5+5, T = 146097, D = 2451545;

int q;

struct Node{
	ll r;
	int id;
	ll dy, mh, yr;
	int tp;
} qes[Q];

int mh[T+5], dy[T+5], yr[T+5];//暴力求出一个周期内的情况 

inline bool cmp1(Node x, Node y) {
	return x.r < y.r;
}

inline bool cmp2(Node x, Node y) {
	return x.id < y.id;
}

inline ll get(ll x, int op) {
	if (op == -1) {
		if (x % 4 == 1) return 366;
		else return 365;
	}
	else if (x < 1582){
		if (x % 4 == 0) return 366;
		else return 365;
	}
	else if (x == 1582) {
		return 355;
	}
	else {
		if ((x % 400 == 0) || (x % 4 == 0 && x % 100 != 0)) return 366;
		else return 365;
	}
}

inline void calc(int p, ll t, ll y, ll op) {
	qes[p].yr = y, qes[p].tp = op;
	if (t <= 31){
		qes[p].dy = t, qes[p].mh = 1;
		return;
	}
	t -= 31;
	int flag = (get(y, op) == 366);
	if (t <= 28 + flag) {
		qes[p].dy = t, qes[p].mh = 2;
		return;
	}
	t -= (28 + flag);
	if (t <= 31) {
		qes[p].dy = t, qes[p].mh = 3;
		return;
	}
	t -= 31;
	if (t <= 30) {
		qes[p].dy = t, qes[p].mh = 4;
		return;
	}
	t -= 30;
	if (t <= 31) {
		qes[p].dy = t, qes[p].mh = 5;
		return;
	}
	t -= 31;
	if (t <= 30) {
		qes[p].dy = t, qes[p].mh = 6;
		return;
	}
	t -= 30;
	if (t <= 31) {
		qes[p].dy = t, qes[p].mh = 7;
		return;
	}
	t -= 31;
	if (t <= 31) {
		qes[p].dy = t, qes[p].mh = 8;
		return;
	}
	t -= 31;
	if (t <= 30) {
		qes[p].dy = t, qes[p].mh = 9;
		return;
	}
	t-= 30;
	flag = (get(y, op) == 355); //1582
	if (flag) {
		if (t <= 21) {
			qes[p].mh = 10;
			if (t <= 4) qes[p].dy = t;
			else qes[p].dy = t + 10;
			return;
		}
		t -= 21;
	}
	else {
		if (t <= 31) {
			qes[p].dy = t, qes[p].mh = 10;
			return;
		}
		t -= 31;
	}
	if (t <= 30) {
		qes[p].dy = t, qes[p].mh = 11;
		return;
	}
	t -= 30;
	qes[p].dy = t, qes[p].mh = 12;
}

inline void work() {
	ll y = 4713, sum = 0;
	
	int p = 1;
	while (y > 0) {
		ll t = get(y, -1);
		if (sum + t < qes[p].r) {
			y--;
			sum += t;
		}
		else {
			calc(p, qes[p].r - sum, y, -1);
			p++;
		}
		if (p > q) return;
	}
	
	y = 1;
	while (p <= q && y <= 1999) {
		ll t = get(y, 1);
		if (sum + t < qes[p].r) {
			y++;
			sum += t;
		}
		else {
			calc(p, qes[p].r - sum, y, 1);
			p++;
		}
	}
	
	y = 2000;
	int x = 1, z = 1;
	while (y < 2400) {
		calc(0, z, y, 1);
		yr[x] = qes[0].yr, mh[x] = qes[0].mh, dy[x] = qes[0].dy;
		x++, z++;
		if (get(y, 1) == 365) {
			if (z == 366) z = 1, y++;
		}
		else {
			if (z == 367) z = 1, y++;
		}
	}
	
	while (p <= q) {
		ll a = (qes[p].r - D) / T, b = (qes[p].r - D) % T;
		if (b == 0) b = T, a--;
		qes[p].yr = a * 400 + yr[b];
		qes[p].mh = mh[b];
		qes[p].dy = dy[b];
		qes[p].tp = 1;
		p++;
	}
}

int main()
{
	scanf("%d", &q);
	for (int i = 1; i <= q; i++) {
		scanf("%lld", &qes[i].r);
		qes[i].id = i;
		qes[i].r++;
	}
	
	sort(qes+1, qes+1+q, cmp1);
	
	work();
	
	sort(qes+1, qes+1+q, cmp2);
	
	for (int i = 1; i <= q; i++) {
		printf("%lld %lld %lld", qes[i].dy, qes[i].mh, qes[i].yr);
		if (qes[i].tp == -1) {
			printf(" BC\n");
		}
		else {
			printf("\n");
		}
	}
	
	return 0;
}

B 动物园

题意

原题：[CSP-S2020] 动物园

分析

稍加分析可以发现，就是确定二进制数的位数以及某些位必须为 000 ，求二进制数个数。

又因为 aia_iai​ 和 qiq_iqi​ 均互不相同，所以不需要 log⁡\loglog 处理。

答案就是 2自由位的数量−n2^{\text{自由位的数量}}-n2自由位的数量−n 。

需要注意位移的时候要 1ull1ull1ull ，还有 k=64,n=0k=64,n=0k=64,n=0 无法用 ull 表示，需要直接 puts("18446744073709551616") ，以及若要计算 264−n2^{64}-n264−n 可以直接用 0ull-(ull)n 自然溢出，不然左移 646464 是 UB。

源码

int n, m, c, k, t;
ull a, b;

int main()
{
	scanf("%d%d%d%d", &n, &m, &c, &k);
	for (int i = 1; i <= n; i++) {
		ull x;
		scanf("%llu", &x);
		a |= x;
	}
	for (int i = 1; i <= m; i++) {
		int p, q;
		scanf("%d%d", &p, &q);
		if (p >= k) continue;
		if (b & (1ull<<p)) continue;
		b |= (1ull<<p);
		if ((a & (1ull<<p)) == 0) t++;
	}
	k = k - t;
	if (k == 64) {
		if (n == 0) printf("18446744073709551616");
		else printf("%llu", 0ull - (ull)n);
	}
	else printf("%llu", (1ull << k) - (ull)n);
	
	return 0;
}

C 函数调用

题意

原题：[CSP-S2020] 函数调用

分析

首先可以把初始值看做 nnn 个加法操作。

对于乘法操作，相当于对之前的操作乘上一个叠乘系数，并且这些系数会乘到加法操作上。

所以容易想到这个问题需要倒序处理。

此外也容易想到把操作均看成图上的点，操作 333 就是在连有向边。

那么可以先建立一个超级源点，指向初始值的 nnn 个操作，然后指向输入的 QQQ 个操作。

我们把图建出来，由题意这一定是个 DAG 。这时候可以递归计算出 fif_ifi​ ，表示 iii 号点的叠乘系数。显然对于加法操作点 fi=1f_i = 1fi​=1，乘法操作点 fi=vif_i=v_ifi​=vi​，有出边的操作点即操作 333 的点 fi=∏i→jfjf_i = \prod\limits_{i\rightarrow j}{f_j}fi​=i→j∏​fj​ 。

再记 CiC_iCi​ 表示 iii 号点的执行次数。

考虑一个操作 333 的点 xxx ，把它指向的节点按输入顺序排序，然后后面的点的 fif_ifi​ 将会对前面点的 CiC_iCi​ 产生贡献。也即对于一个点 iii 的调用次数，有 Ci=∑x→i{Cx×∏x→j&&j is after ifj}C_i =\sum\limits_{x\rightarrow i} \left\{C_x \times \prod\limits_{x\rightarrow j \&\& \text{j is after i}}{f_j}\right\}Ci​=x→i∑​{Cx​×x→j&&j is after i∏​fj​} （此处 →\rightarrow→ 意为有边指向。）

这个东西可以拓扑排序的时候做。

又注意到用前向星存图的话，直接把出边顺序翻转了，所以只要顺着处理下来，就相当于是反着在统计上述的 fjf_jfj​ ，刚好与意义吻合。

需要注意区别 fif_ifi​ 和 CiC_iCi​ ，一个乘法点的操作数 CiC_iCi​ 并不一定等于它对之前操作的系数贡献 fif_ifi​ 。fif_ifi​ 在图拓扑排序之前就已经确定，只与它之后的节点有关。但 CiC_iCi​ 是与之前的节点有关，需要拓扑的时候 DP 计算。

此外有个细节，拓扑排序需要把所有入度为 000 的点入队。不能想着只有超级源点有价值就只入超级源点。因为别的点也对其它点有入度贡献，不如队会导致有的点入度降不到 000 。

源码

const int N = 1e5+5, M = 2e6+5, mod = 998244353;
typedef long long ll;
#define gc getchar

inline int rd() {
	int x = 0, fg = 1;
	char ch = gc();
	while (ch < '0' || ch > '9') {
		if (ch == '-') fg = -1;
		ch = gc();
	}
	while (ch >= '0' && ch <= '9') {
		x = (x << 1) + (x << 3) + ch - '0';
		ch = gc();
	}
	return x * fg;
}

int n, m, Q;
ll a[N];
struct OP {
	int tp, p, v;
} op[N<<1];

int d[N<<1];//入度 
int g[N<<1], v[M], nxt[M], tot;//前向星存图, 遍历时刚好为逆序, 便于计算 
inline void add(int x, int y) {
	v[++tot] = y, nxt[tot] = g[x], g[x] = tot, d[y]++;
}

int q[N<<1], hd, tl;
ll c[N<<1], f[N<<1];//每个操作要执行的次数 每个操作之前的操作的叠乘系数 
void work() {
	hd = tl = 0;
	for (int i = 0; i <= n+m; i++) if (d[i] == 0) q[tl++] = i;//不能只加入超级源点, 因为其他入度为0的点也可能对有用的点有入度贡献，不加入进去有的点入度到不了0 
	while (hd != tl) {
		int x = q[hd++];
		if (op[x].tp == 1) a[op[x].p] = (a[op[x].p] + c[x] * op[x].v % mod) % mod;
		else if (op[x].tp == 3) {
			ll t = c[x];
			for (int i = g[x]; i; i = nxt[i]) {
				int y = v[i];
				c[y] = (c[y] + t) % mod;
				t = t * f[y] % mod;
				if (--d[y] == 0) q[tl++] = y;
			}
		}
	}
}

void calc(int x) {//处理叠乘系数 
	if (f[x] != -1) return;//要么不是 3 操作, 要么计算过了不能重复计算 
	f[x] = 1;
	for (int i = g[x]; i; i = nxt[i]) {
		int y = v[i];
		calc(y);
		f[x] = f[x] * f[y] % mod;
	}
}

int main()
{
	n = rd();
	for (int i = 1; i <= n; i++) //把初始值也转化为操作 
		op[i].v = rd(), op[i].p = i, op[i].tp = 1, f[i] = 1, add(0, i);
	m = rd();
	for (int i = n+1; i <= n+m; i++) {
		op[i].tp = rd();
		if (op[i].tp == 1) op[i].p = rd(), op[i].v = rd(), f[i] = 1;
		else if (op[i].tp == 2) op[i].v = rd(), f[i] = op[i].v;
		else {
			int t = rd();
			for (int j = 1; j <= t; j++) add(i, rd() + n);
			f[i] = -1;
		}
	} 
	
	for (int i = 1; i <= n+m; i++) calc(i);
	
	Q = rd();
	for (int i = 1; i <= Q; i++) add(0, rd() + n);
	op[0].tp = 3, c[0] = 1;
	
	work();
	
	for (int i = 1; i <= n; i++) printf("%lld ", a[i]);
	
	return 0;
}

D 贪吃蛇

题意

原题：[CSP-S2020] 贪吃蛇

  • 蛇按权为第一关键字，编号为第二关键字排序。

  • 每次最大的蛇可以选择吃最小的蛇，并把权减去最小的蛇作为新权；或者选择结束游戏。

  • 蛇希望在活着的前提下吃更多的蛇，并且蛇足够聪明。

  • 求最后剩几条蛇。

  • 多组测试。

分析

这是贪心以及博弈综合。

结论：

1.如果当前最大的蛇吃了最小的蛇，没有成为新的最小的蛇，必然选择吃。特殊的，两条蛇的时候必然会选择吃。
2.如果会成为最小的蛇，需要考虑还会经过多少轮到达一个上述 1 的局面。如果相距轮数为偶数则吃，否则不吃并结束。

对于 1，显然可以归纳证明。

  • n=2,3n=2,3n=2,3 时成立。
  • 假设 nnn 条蛇的时候成立。新加入第 n+1n+1n+1 条蛇后，记最大、次大、次小、最小的蛇分别为 a,b,c,da,b,c,da,b,c,d 。即要证明 c<a−dc<a-dc<a−d 时 a−da-da−d 不会被吃掉。由于 nnn 条蛇时命题成立，所以要么 bbb 选择不吃，结束游戏，此时 n+1n+1n+1 的命题成立；要么 bbb 选择吃 ccc，且必然满足 b−cb-cb−c 不会被吃掉。由于 a−d>b−ca-d>b-ca−d>b−c ，所以 a−da-da−d 也必然不会被吃掉，所以命题成立。

对于 2，考虑达到 1 局面的时候为第 kkk 轮，此时最小的蛇必定会被吃掉。那么在 k−1k-1k−1 轮的时候，最大的蛇若选择吃，就会变成最小的蛇，进而在第 kkk 轮被吃掉，所以它不会选择吃。因此 k−2k-2k−2 轮的最大蛇不会因为变成最小的蛇而在第 k−1k-1k−1 轮被吃掉，所以它可以吃，因此他会选择吃。同理往上推就可以了。

所以我们需要一个数据结构，每次可以取出最大值和最小值，并插入新值。配合这个数据结构模拟即可。

由于 n≤106n \le 10^6n≤106 并且 T≤10T \le 10T≤10 ，这个数据结构不能带 log⁡\loglog ，必须是线性的。

引理：

后吃的蛇吃之后变成的蛇 必定小于 先吃的蛇吃之后变成的蛇。（因为有第二关键字在，这里不考虑等号）

可以反证证明。

考虑 a>b>c>d>ea>b>c>d>ea>b>c>d>e 。必然有 aaa 吃 eee 变成 a−ea-ea−e 。

  • 若 a−e>ba-e>ba−e>b ，最大蛇还是 aaa ，必然满足引理。
  • 若 b>a−e>cb>a-e>cb>a−e>c，则下一轮 bbb 吃 ddd 变成 b−db-db−d ，显然 a−e>c>b−da-e>c>b-da−e>c>b−d ，最大的蛇还是 aaa，继续进行下去无法推翻引理。
  • 若 c>a−e>dc > a-e > dc>a−e>d ，下一轮 bbb 吃 ddd 变成 b−db-db−d ，局面变成 c>a−e>b−dc>a-e>b-dc>a−e>b−d 。下一轮有 ccc 吃 b−db-db−d 变成 c−b+dc-b+dc−b+d。假设不满足引理，即 c−b+d>a−ec-b+d>a-ec−b+d>a−e 。由于 c−b<0c-b<0c−b<0，故有 d>a−ed>a-ed>a−e 。但这种情况是满足 a−e>da-e>da−e>d 的，矛盾，故假设不成立，原命题成立。
  • 若 d>a−ed > a-ed>a−e ，aaa 成了最小的蛇，根据贪心结论，它不会被吃掉，游戏结束，故满足引理。

综上，引理成立。

所以我们可以用类似 [NOIP2016]蚯蚓 的方法，开两个队列，在维护时保证分别单增。这可以 O(1)O(1)O(1) 取出全局最大值最小值。具体地，先把整个数列丢进 q1q_1q1​，每次吃出的新蛇从队头丢进 q2q_2q2​ 。由引理知 q2q_2q2​ 继续单增。因此可以线性维护。

如果要减小常数可以手写队列，但本人手写之后感觉这里直接用 deque 更方便。

其它细节见代码。

源码

#define gc getchar
#define mkp make_pair
#define fi first
#define se second
typedef pair<int,int> pii; 
const int N = 1e6+5;
const pii inf = mkp(1e9+5, 1e6+5), ind = mkp(-1e9-5, 0);

inline int rd() {
	int x = 0, fg = 1;
	char ch = gc();
	while (ch < '0' || ch > '9') {
		if (ch == '-') fg = -1;
		ch = gc();
	}	
	while (ch >= '0' && ch <= '9') {
		x = (x << 1) + (x << 3) + ch - '0';
		ch = gc();
	}
	return x * fg;
}

int n;
int a[N];

pii q1[N], q2[N];//维护时保证两个队列单增 
int hd1, tl1, hd2, tl2;

inline int add(int x) { if (++x == N) x = 0; return x; }
inline int sub(int x) { if (--x == -1) x = N-1; return x; }
inline pii mx1() { return sub(hd1) != tl1 ? q1[tl1] : ind; }
inline pii mn1() { return sub(hd1) != tl1 ? q1[hd1] : inf; }
inline pii mx2() { return sub(hd2) != tl2 ? q2[tl2] : ind; }
inline pii mn2() { return sub(hd2) != tl2 ? q2[hd2] : inf; }

int solve() {
	hd1 = hd2 = 0, tl1 = tl2 = N-1;
	int num = n, sum = n;//num为真实的蛇的数量, sum 为假设可以继续进行的蛇数量 
	for (int i = 1; i <= n; i++) q1[tl1 = add(tl1)] = mkp(a[i], i);
	int flag = 0, cur = 0;//是否进入第二阶段, 进入轮数的奇偶 
	while (sum >= 2) {
		pii mx, mn;
		if (mx1() < mx2()) mx = mx2(), tl2 = sub(tl2);
		else mx = mx1(), tl1 = sub(tl1);
		if (mn1() > mn2()) mn = mn2(), hd2 = add(hd2);
		else mn = mn1(), hd1 = add(hd1);
	
		mx.fi -= mn.fi;
		sum--;
		
		if (sum >= 2 && mx < min(mn1(), mn2())) {
			if (flag) cur ^= 1;
			else flag = 1;
		}
		else {
			if (flag) return num - cur;
			num--;
		}
		q2[hd2 = sub(hd2)] = mx;
	}
	return num;
}

int main()
{
	int T = rd();
	for (int i = 1; i <= T; i++) {
		if (i == 1) {
			n = rd();
			for (int j = 1; j <= n; j++) a[j] = rd();
		}
		else {
			int k = rd();
			for (int j = 1; j <= k; j++) a[rd()] = rd();
		}
		printf("%d\n", solve());
	}
	return 0;
}
