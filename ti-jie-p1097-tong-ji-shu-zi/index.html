题解 P1097 【统计数字】

2019-09-08 21:07 2 min read # 题解

刚拿到这道题被10^9惊艳到了，想了一大堆离散化什么的，做着做着突然意识到，直接把输入排列一遍，再O(n)遍历，一边统计一边输出就可以了！

考虑用快排的时候发现sort并不会超时，于是直接sort就结束了。。

#include <iostream>
#include <algorithm>

using namespace std;

const int maxn = 200005;

int n;
long long a[maxn];

int main()
{
	cin>>n;
	for(int i = 0; i < n; i++){
		cin>>a[i];
	}
	
	sort(a, a+n);
	
	int sum = 0;
	for(int i = 1; i < n; i++){
		sum++;
		if(a[i]!=a[i-1]){//一旦两数不同，就说明等于a[i-1]的数已经没了，所以直接输出sum，再将其重置 
			cout<<a[i-1]<<' '<<sum<<endl;
			sum = 0;
		}
	}
	
	sum++;//因为在循环里面最后一个数并没有使sum自增 
	cout<<a[n-1]<<' '<<sum<<endl;
	
	return 0;
}
