公平组合游戏ICG

公平组合游戏 (Impartial Combinatorial Games) :

  • 两名选手交替进行游戏
  • 任一操作只与游戏本身有关，与操作选手等其他因素无关
  • 不存在平局

博弈论中，有一些有代表性的 ICG ，和一些常用方法。

Nim游戏

定义

通常的Nim游戏的定义是这样的：有若干堆石子，每堆石子的数量都是有限的，合法的移动是“选择一堆石子并拿走若干颗（不能不拿）”，如果轮到某个人时所有的石子堆都已经被拿空了，则判负（因为他此刻没有任何合法的移动）。

结论

记各堆石子数分别为 a1,a2,...,ana_1, a_2, ... , a_na1​,a2​,...,an​ 。
先手必胜，当且仅当 XOR⁡ ai≠0\operatorname{XOR} \, a_i \neq 0XORai​​=0

考虑证明。

当没有石子时，显然先手必败，一次性拿完即可。

考虑 XOR⁡ ai=0\operatorname{XOR} \, a_i = 0XORai​=0 的局面。
此时一定不存在一种拿法，使得拿完之后其异或和仍为0。
假设存在，把 a1a_1a1​ 拿完变成 a1′a_1'a1′​。那么有 a1′XOR⁡a2XOR⁡...XOR⁡an=0a_1' \operatorname{XOR} a_2 \operatorname{XOR} ... \operatorname{XOR} a_n = 0a1′​XORa2​XOR...XORan​=0
则可知 a1=a1′a_1 = a_1'a1​=a1′​，相当于没有拿，这是不合法的。
故一定不存在这种拿法。
故先手会一直因后手的操作，保持在异或和为 000 的局面，直至最后没有石子。
故先手必败。

换言之，先手必胜，当且仅当 XOR⁡ ai≠0\operatorname{XOR} \, a_i \neq 0XORai​​=0。

有向图游戏

给定一个 有向无环图 和一个起始顶点上的一枚棋子，两名选手交替的将这枚棋子沿有向边进行移动，无法移 动者判负。事实上，这个游戏可以认为是 所有公平组合游戏的抽象模型。

解决这种问题，常用 SG 函数

SG函数

定义

记 g(x)g(x)g(x) 为点 xxx 的 SG 函数。引入 mex⁡\operatorname{mex}mex 运算 ，mex⁡{a1,a2,...,an}\operatorname{mex}\{ a_1, a_2, ... , a_n\}mex{a1​,a2​,...,an​} 的结果为此集合中未出现的最小自然数。

那么 g(x)=mex⁡{g(ui)}g(x) = \operatorname{mex} \{g(u_i)\}g(x)=mex{g(ui​)} ，其中 uiu_iui​ 为 xxx 所指向的节点。

结论

对于上述有向图游戏，记起点为 sss ，那么 先手必胜当且仅当 g(s)≠0g(s) \neq 0g(s)​=0 。根据每步操作带来的变化，可以用类似于 Nim游戏 的证明方法加以证明。

对于多个有无环向图，则先手必胜当且仅当 XOR⁡g(si)≠0\operatorname{XOR} g(s_i) \neq 0XORg(si​)​=0 。

求法

对于一个有向无环图，dfs 即可计算求解 SG 函数。

acoj - 12754 - 移棋子游戏

/**********************************

Problem: acoj - 12754 - 移棋子游戏

From: acoj

Algorithm: 博弈论 

Author: cyh_toby

Last updated on 2020.5.24

**********************************/
#include <cstdio>
#include <cstring>
#include <set>

using namespace std;

const int N = 2005, M = 6005;

int g[N], nxt[M], v[M], idx;
int n, m, k;
int f[N];//SG
int ans = 0;

inline void add(int x, int y) {
	idx++;
	v[idx] = y;
	nxt[idx] = g[x];
	g[x] = idx;
	return;
}

int dfs(int u) {//求u的SG值 
	if (f[u] >= 0)
		return f[u];
	
	f[u] = 0;
	set<int> s;
	for (int i = g[u]; i; i = nxt[i])
		s.insert(dfs(v[i]));
	
	for (int i = 0; ; i++) {
		if (!s.count(i)) {
			f[u] = i;
			break;
		}
	}
	
	return f[u];
}

int main()
{
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 1; i <= m; i++) {
		int a, b;
		scanf("%d%d", &a, &b);
		add(a, b);
	}
	
	memset(f, -1, sizeof(f));
	
	for (int i = 1; i <= k; i++) {
		int x;
		scanf("%d", &x);
		ans ^= dfs(x);
	}
	
	if (ans)
		puts("win");
	else
		puts("lose");
	
	return 0;
}

acoj12755牛的数字游戏

acoj - 12755 - 牛的数字游戏

把每个数字拆分开，按题意模拟建图，使用 SG 函数。

/**********************************

Problem: acoj - 12755 - 牛的数字游戏 

State: Accepted

From: acoj

Algorithm: 博弈论, SG函数 

Author: cyh_toby

Last updated on 2020.5.31

**********************************/
#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 1000005;

int f[N];

inline void getm(int x, int &r1, int &r2) {
	r1 = 0, r2 = 10;
	while (x) {
		r1 = max(r1, x % 10);
		r2 = min(r2, x % 10 == 0 ? 10 : (x % 10));
		x /= 10;
	}
	return;
}

inline void init() {
	f[0] = 0;
	for (int i = 1; i <= N-5; i++) {
		int x, y;
		getm(i, x, y);
		x = i - x, y = i - y;
		x = f[x], y = f[y];
		if (x < y)
			swap(x, y);
			
		if ((x == 0 && y == 0) || (x == 2 && y == 0))
			f[i] = 1;
		if (y > 0)
			f[i] = 0;
		if (x == 1 && y == 0)
			f[i] = 2;
	}
	return;
}

int main()
{
	init();
	
	int T;
	scanf("%d", &T);
	while (T--) {
		int n;
		scanf("%d", &n);
		if (f[n])
			puts("YES");
		else
			puts("NO");
	}
	return 0;
}

acoj12756取石子游戏

acoj - 12756 - 取石子游戏

与上一题类似，根据取石子的限制条件建图（可以不用真的建出来），随后 SG 求解。

/**********************************

Problem: acoj - 12756 - 取石子游戏 

State: Accepted

From: acoj

Algorithm: 博弈论, SG函数 

Author: cyh_toby

Last updated on 2020.5.31

**********************************/
#include <cstdio>
#include <set>
#include <cstring>

using namespace std;
const int N = 15, K = 1005;

int n, m, a[N], b[N], f[K];
int book[K];

inline void init() {
	f[0] = 0;
	for (int i = 1; i < K; i++) {
		memset(book, 0, sizeof(book));
		for (int j = 1; j <= m; j++) {
			if (i < b[j]) break;
			book[f[i-b[j]]] = 1;
		}
		for (int j = 0; ; j++) {
			if (!book[j]) {
				f[i] = j;
				break;
			}
		}
	}
	return;
}

int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	scanf("%d", &m);
	for (int i = 1; i <= m; i++)
		scanf("%d", &b[i]);
	
	init();
	
	int res = 0;
	for (int i = 1; i <= n; i++)
		res ^= f[a[i]];
	
	if (res)
		puts("YES");
	else
		puts("NO");
	
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			if (a[i] - b[j] >= 0 && ((res ^ f[a[i]] ^ f[a[i] - b[j]]) == 0)) {
				printf("%d %d\n", i, b[j]);
				return 0;
			}
		}
	}
	
	return 0;
}
