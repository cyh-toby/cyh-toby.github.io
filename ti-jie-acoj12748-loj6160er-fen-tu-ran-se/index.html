分析

将二分图转化为矩阵分析。

显然绿色无关紧要，剩下的格子默认为绿色即可。转化为只放红色或蓝色。

Part 2: 容斥

fnf_nfn​ 存储的是一种颜色的放置方案数；现欲计算两种颜色，考虑容斥。

  • 直接 fn×fnf_n \times f_nfn​×fn​ ，多出了一些方案：两种颜色重叠放置在同一格子。

  • 减去两种颜色重叠的方案数。总共有 Cn1×An1C_n^1 \times A_n ^1Cn1​×An1​ 个格子可能用来重叠，则需减去 Cn1×An1×fn−1    2C_n^1 \times A_n^1 \times f_{n-1}^{\;\;2}Cn1​×An1​×fn−12​

#include <cstdio>

using namespace std;

const int mod = 1000000007, N = 10000005;
typedef long long ll;
int n, fac[N], infac[N], f[N];

inline ll qpower(ll a, ll b, ll p) {
	ll res = 1ll;
	while (b) {
		if (b & 1)
			res = res * a % p;
		a = a * a % p;
		b >>= 1;
	}
	return res;
}

inline void init() {
	fac[0] = 1;
	for (int i = 1; i <= n; i++)
		fac[i] = 1ll * fac[i-1] * i % mod;
	infac[n] = qpower(fac[n], mod-2, mod);
	for (int i = n-1; i >= 0; i--)
		infac[i] = 1ll * infac[i+1] * (i+1) % mod;
		
	f[0] = 1, f[1] = 2;
	for (int i = 2; i <= n; i++) 
		f[i] = (2ll * i % mod * f[i-1] % mod - 1ll * (i-1) * (i-1) % mod * f[i-2] % mod + mod) % mod;
	return;	
}

inline int C(int a, int b) {
	if (a < b) return 0;
	return 1ll * fac[a] * infac[b] % mod * infac[a-b] % mod;
}

inline int A(int a, int b) {
	if (a < b) return 0;
	return 1ll * fac[a] * infac[a-b] % mod;
}

int main()
{	
	scanf("%d", &n);
	init();
	
	ll ans = 0;
	for (int i = 0; i <= n; i++) 
		if (i % 2 == 0)
			ans = (ans + 1ll * C(n, i) * A(n, i) % mod * f[n-i] % mod * f[n-i] % mod) % mod;
		else
			ans = (ans - 1ll * C(n, i) * A(n, i) % mod * f[n-i] % mod * f[n-i] % mod) % mod;
	
	printf("%lld", (ans + mod) % mod);
	
	return 0;
}
