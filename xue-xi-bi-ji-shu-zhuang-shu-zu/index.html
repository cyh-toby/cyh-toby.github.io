单点修改区间查询

/**********************************

Problem: luogu - P3374 - 【模板】树状数组（单点修改区间查询） 

State: Accepted

From: 【模板】 

Algorithm: 树状数组 

Author: cyh_toby

last updated on 2020.3.11

**********************************/
#include <cstdio>
using namespace std;
const int N = 500005;
int n, m;
int c[N];
inline void updat(int x, int v) {
	for (; x <= n; x += x & (-x))
		c[x] += v;
	return;
}
inline int query(int x) {
	int res = 0;
	for (; x; x -= x & (-x))
		res += c[x];
	return res;
}
int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		int a;
		scanf("%d", &a);
		updat(i, a);
	}
	for (int i = 1; i <= m; i++) {
		int op, x, y;
		scanf("%d%d%d", &op, &x, &y);
		if (op == 1) 
			updat(x, y);
		else
			printf("%d\n", query(y) - query(x-1));
	}
	return 0;
}

区间修改单点查询

/**********************************

Problem: luogu - P3368 - 【模板】树状数组(区间修单查) 

State: Accepted

From: 【模板】 

Algorithm: 树状数组 

Author: cyh_toby

last updated on 2020.3.11

思想：建立差分数组，化区间修改为单点修改，用树状数组维护差分数组前缀和，即原数组单点查询 
**********************************/
#include <cstdio>
using namespace std;
const int N = 500005;
int n, m;
int c[N];//无需建立差分数组，只用维护差分数组前缀和的树状数组 
inline int query(int x) {
	int res = 0;
	for (; x; x -= x & (-x))
		res += c[x];
	return res;
}
inline void updat(int x, int v) {
	for (; x <= n; x += x & (-x))
		c[x] += v;
	return;
}
int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) {
		int a;
		scanf("%d", &a);
		updat(i, a);
		updat(i+1, -a);
	}
	for (int i = 1; i <= m; i++) {
		int op, x, y, k;
		scanf("%d", &op);
		if (op == 1) {
			scanf("%d%d%d", &x, &y, &k);
			updat(x, k);
			updat(y+1, -k);
		}
		else {
			scanf("%d", &x);
			printf("%d\n", query(x));
		}
	}	
	return 0;
}
