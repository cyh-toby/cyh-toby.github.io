习题日志-19.5.29

2019-05-29 20:53 2 min read # 习题日志

题目：油田uva572

知识点：dfs,图(很简单的图的应用)

源码：

#include <iostream>

using namespace std;

#define maxn 105 

int m, n;
char pic[maxn][maxn];
int idx[maxn][maxn];

void setagain()
{
	for (int i = 0; i < maxn; i++){
		for (int j = 0; j < maxn; j++){
			idx[i][j] = 0;
		}
	}
	return;
}

void dfs(int r, int c, int id){
	if (r < 0 || r >= m || c < 0 || c >= n)
		return;
	if (idx[r][c] > 0 || pic[r][c] != '@')
		return;
	
	idx[r][c] = id;
	for (int i = -1; i <= 1; i++){
		for (int j = -1; j <= 1; j++){
			dfs(r+i, c+j, id);
		}
	}
	
	return;
}

int main()
{
	while (1){
		setagain();
		int num = 0;
		cin>>m>>n;
		if (m==0)
			break;
		
		for (int i = 0 ; i< m; i++){
			for (int j = 0; j < n; j++){
				cin >>pic[i][j];
			}
		}
		
		for (int i = 0 ; i < m; i++){
			for (int j = 0 ;j < n; j++){
				if (idx[i][j]==0 && pic[i][j] == '@'){
					num++;
					dfs(i, j , num);
				}
				
			}
		}
		
		cout<<num<<endl;
		
	}
	return 0;
}

心得：dfs的简单应用，几乎不涉及图的理论。调试时出错就在于没有写setagain()函数（初始化），导致第二个和后面结果不对，要加强对处理多组数据需重置的敏感度。
