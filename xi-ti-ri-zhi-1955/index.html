题目:树的层次遍历(uva122)

知识点:bfs、队列、树

代码：

#include <iostream>
#include <queue>
#include <string>
#include <cstdio>//sscanf
#include <cstring>

using namespace std;

#define maxn 100000

int nsize = 1;
bool allok = true;
int ans[maxn];

struct node{
	bool have_value;
	int value;
	int left;
	int right;
};
node tree[maxn];

void setagain(){
	for (int i = 0; i < maxn;i++){//可优化 
		tree[i].left = -1;
		tree[i].right = -1;
		tree[i].value = -1;
		tree[i].have_value = false;
	}
	nsize = 1;
	allok = true;
}

bool read()
{
	char s[maxn];
	char s1[maxn];
	int v = 0;
	int u = 0;
	
	bool have_reads = false;
	while (scanf("%s", s)){
		have_reads = true;
		if (strlen(s) == 2){
			break;
		}	
		sscanf(&s[1], "%d", &v);
		sscanf(strchr(s, ',')+1, "%s", &s1);//IMPORTANT!sscanf( &+name, "%+sth", &+name);格式注意：前加& 
		u = 0;	
		for (int i = 0; i < strlen(s1)-1;i++){
			if (s1[i] == 'L'){
				if (tree[u].left == -1){
					tree[u].left = nsize++;
				}
				u = tree[u].left;
			}
			if (s1[i] == 'R'){
				if (tree[u].right == -1){
					tree[u].right = nsize++;
				}
				u = tree[u].right;
			}
		}
		
//		cout << u << " "<< v<<endl;
		tree[u].value = v;
		if (!tree[u].have_value)//没有value 
		{
			
			tree[u].have_value = true;
		}
		else{
			allok = false;//allok here is changeing
		}
	}
	
	if (have_reads){
		return true;
	}	
	else{
		return false;
	}
}

void bfs(){
	queue <int> q;//队列 
	q.push(0);
	int p = 0;
	while (!q.empty()){
		int u = q.front();
		q.pop();
		if (tree[u].value != -1){
			ans[p++] = tree[u].value;
		}
		else{
			allok = false;
			return;
		}
		
		if (tree[u].left != -1){
			q.push(tree[u].left);
		}
		if (tree[u].right != -1){
			q.push(tree[u].right);
		}
	}
	return;
}

int main()
{	
	setagain();	//初始化不能在循环一次后再实施 也不能放一开始，应放最后 
	while (read()){
		
		bfs();
		
		if (allok == false){
			cout << "not complete"<<endl;
			continue;
		}
		
		for (int i = 0; i < nsize;i++){
			cout << ans[i]<<" ";
		}
		cout<<endl;
		
		setagain();
	}
	return 0;
}

心得：本题难度较大，作为本人首次接触的队列题，难以攻克的却是输入及用数组模拟树。花了将近共4个多小时才解出，但问题主要出在几个很弱智低级的错误，以下给予自己今后做题以提醒：

  1. 只用一次的代码不要手残放到循环内，指不定会发生什么；
  2. 没有初始化或初始化了却没赋予应有值的变量不可轻易使用，尤其是作为for的判断条件;
  3. /<stdio.h>下的sscanf( ，，)函数一三两项是地址而不是变量名 eg.sscanf(&s, "%s", &s1);//其中，s and s1都必须为char类型变量/数组
  4. 用途为初始化的代码/函数注意位置，循环外应使用一次，循环内应放最后；

最后，贴上一些新学的函数：

  • :queue 变量名 --队列
  • :sscanf( , , ) --从变量读到变量
  • :strchr(数组名， 字符）--char数组中此字符的位置
  • :strlen(数组名） --char数组的长度
