原题：acwing143(关于Trie的应用题)

源码：

#include <bits/stdc++.h>

using namespace std;

#define maxn 100005

int n;

int trie[32*maxn][2];
int endt[32*maxn];
int s[35];
int a[maxn];
int ans = 0;
int tot = 1;
inline void insert_trie(int str[], int x)
{
	int p = 1;
	int len = 32;
	for(int i = 0; i < len; i++){
		int c = str[i];
		if(!trie[p][c]) trie[p][c] = ++tot;
		p = trie[p][c];
	}	
	endt[p] = x;
	return;
}
inline int search_trie(int str[]){
	int p = 1;
	int len = 32;
	for(int i = 0; i < len; i++){
		int c = (str[i]==1?0:1);
		if(trie[p][c]){
			p = trie[p][c];
			continue;
		}
		else{
			c = (c==1?0:1);
			p = trie[p][c];
			continue;
		}	
	}
	return endt[p];
}

int main() 
{
	scanf("%d", &n);
	for(int i = 1; i <= n; i++){
//		int a[i];
		scanf("%d", &a[i]);
		int tt = a[i];
		for(int j = 32-1; j >= 0; j--){
			if(tt==0){
				s[j] = 0;
				continue;
			}
			s[j] = tt&1;
			tt>>=1;
		}
		insert_trie(s, i);
		int t = search_trie(s);
		ans = max(ans, a[t]^a[i]);
//		printf("%d = (", a[i]);
//		for(int k = 0; k <= 31; k++){
//			printf("%d", s[k]);
//		}
//		printf(")2\n");
	}

	printf("%d\n", ans);
	
	return 0;
}

之所以专门写这个习题日志，是因为在本题犯了一个不可饶恕的错误 !!!!!!

就是 mainmainmain 函数的第三行 :::

int a[i];

原因是之前打算每次读一个intintint不用数组，后来发现存起来更好，便手抖加了个 [i][i][i]

关键在于，这个错很玄幻，以至于在本地跑 有时 是对的，但是如果中间加入打印语句，结果又不正确了 !!!

这意味着，这种错很容易被忽略，检查不出来 !!!!!!!!!

因此特写次日志引以为戒。

切记不可再随便在已有代码上做如下操作：

  • 加数组下标

  • 加变量声明类型（intintint 、charcharchar 等）

  • 对于链表这类数据结构或算法，在已有代码中间插入新代码打乱原有顺序

  • 可能还有别的，遇见了再继续总结
