题目：机器翻译

知识点：队列？

分析：第一次提交7个WA，主要是本地RUN的时候没有打印出程序的中间结果，只是结果恰巧正确，因此直接测评并没检查出错误。后来活学活用使用队列顺利解决了内存第一项的元素。

源码第一次：

#include <iostream>

using namespace std;

int m, n;
int a[1000];
bool ok[1000] = {0};
int ans = 0;

int main()
{
	cin >> m >> n;
	int first = 0;
	int tm = 0;
	for (int i = 0; i < n; i++){
		cin >> a[i];
	}
	
	for (int i = 0; i < n; i++){
		if(ok[a[i]] == false){
			ans++;
			if (tm < m){
				ok[a[i]] = true;
				tm++;
			}
			else{
				while(ok[a[first]] == false){
					first++;//错大概在这部分吧
				}
				ok[a[first]] = false;
				first++;
				ok[a[i]] = true;
				first++;
			}
		}
		
//		for (int j = 0; j < n; j++){
//			if (ok[j]){
//				cout<<j<<" ";
//			}
//		}
//		cout << endl;
		
	}
	
		
	cout << ans <<endl;
	
	return 0;
}
