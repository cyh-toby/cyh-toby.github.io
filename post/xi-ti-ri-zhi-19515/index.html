题目：树(Uva548)

知识点：树，dfs, stringstream

源码(已AC)：

#include <iostream>
#include <string>
#include <sstream>
#include <algorithm>
#include <climits> 

using namespace std;

const int maxv = 10000+10;

int ans = INT_MAX;
int fans;
int in_order[maxv], post_order[maxv],lch[maxv],rch[maxv];
int  n;

void setagain(){
	ans = INT_MAX;
	for (int i = 0; i < n; i++){
		in_order[i] = post_order[i] = 0;
	}
}

bool read_list(int s){ 
	string line;
	if (!getline (cin, line))
		return false;
	stringstream ss(line);
	n = 0;
	int x;
	while (ss>>x){
		if (s == 1){
			in_order[n++] = x;
		}
		else{
			post_order[n++] = x;
		}
		
	}
	return n>0;
}

int build(int L1, int R1, int L2, int R2){
	if (L1 > R1){
		return 0;
	}
	int root = post_order[R2];
	int r;
	for (int i = L1; i <= R1;i++){
		if (in_order[i] == root){
			r = i;
			break;
		}
	}
	
	lch[root] = build(L1, r-1, L2, L2+r-L1-1);
	rch[root] = build(r+1, R1, L2+r-L1,R2-1);
	
	return root;
}

void dfs(int cur, int sum){
	if (!lch[cur] && !rch[cur]){
		sum += cur;
		if (ans > sum){
			ans = sum;
			fans = cur;
		}
		if (ans == sum && fans > cur){
			fans = cur;
		}
	//	fans = cur;
	//	cout<<"ans: "<<ans<<endl;
	//	cout<<"fans: "<<fans<<endl;
		return;
	}
	
	if (lch[cur]){
		dfs(lch[cur], sum+cur);
	}
	if (rch[cur]){
		dfs(rch[cur], sum+cur);
	}
	if (sum >= ans)
		return;
	
}

int main(){
	while (read_list(1)){
		read_list(2);
	
		build(0, n-1, 0, n-1);

		dfs(post_order[n-1], 0);
	
		cout<<fans<<endl;
		
		setagain();
	}
	return 0;
}

心得：本题主要考察树的构造，dfs只是插曲。具体要点如下。

  1. build 函数的递归语句行值得注意，容易出错，类似的加/减数组下标找位置的代码，需要反复验算后再写。
  2. read_list函数含金量较大，用到下的stringstream，需要巩固复习其用法；在函数括号内定义的数组不知是否就是地址，有待查证eg. void f(int a[]//那此处a是不是传地址呢) 。
    1.最后发现几次本地都跑不对的原因是，题目要求最小和的路径，默认的是包括叶子节点本身的值，自己在写dfs的终止条件时写的是如下，且fans赋值没给判定条件(注释掉的是正确代码)
