题解 [NOIp2012]/P1080 【国王游戏】

2020-03-17 20:38 7 min read # 题解

题意

原题：P1080 国王游戏

  • 给定从 000 开始的长为 n+1n+1n+1 的序列对 a,ba, ba,b ，即 aia_iai​ 与 bib_ibi​ 捆绑在一起。

  • 可进行任意排列。

  • 求 min⁡(max⁡1≤i≤n(1bi×∏j=0i−1aj))\min\Bigg(\max\limits_{1\le i\le n}\Big(\dfrac{1}{b_i}\times\prod\limits_{j=0}^{i-1}a_j\Big)\Bigg)min(1≤i≤nmax​(bi​1​×j=0∏i−1​aj​))

分析

考虑贪心。

首先给出结论：

按 ai×bia_i\times b_iai​×bi​ 升序排列为最优解。

接下来证明。显然可以先从相邻位置开始证明。

比如现考虑 iii 和 i+1i+1i+1 两个位置。

交换这两个之前，他们对答案产生的贡献分别为

s1=max⁡(1bi×∏j=0i−1aj,1bi+1×∏j=0iaj)s_1=\max\bigg(\dfrac{1}{b_i}\times\prod\limits_{j=0}^{i-1}a_j\quad,\quad\dfrac{1}{b_{i+1}}\times\prod\limits_{j=0}^{i}a_j\bigg) s1​=max(bi​1​×j=0∏i−1​aj​,bi+1​1​×j=0∏i​aj​)

交换之后为

s2=max⁡(1bi+1×∏j=0i−1aj,ai+1bi×∏j=0i−1aj)s_2=\max\bigg(\dfrac{1}{b_{i+1}}\times\prod\limits_{j=0}^{i-1}a_j\quad,\quad \dfrac{a_{i+1}}{b_i}\times\prod\limits_{j=0}^{i-1}a_j\bigg) s2​=max(bi+1​1​×j=0∏i−1​aj​,bi​ai+1​​×j=0∏i−1​aj​)

那么，开始分析法。

s1≤s2s_1\le s_2 s1​≤s2​

⇕\Updownarrow ⇕

提取公因式∏j=0i−1aj\text{提取公因式}\prod\limits_{j=0}^{i-1}a_j 提取公因式j=0∏i−1​aj​

max⁡(1bi  ,aibi+1)  ≤  max⁡(1bi+1  ,  ai+1bi)\max\bigg(\dfrac{1}{b_i}\;,\dfrac{a_i}{b_{i+1}}\bigg)\;\le\;\max(\dfrac{1}{b_{i+1}}\;,\;\dfrac{a_{i+1}}{b_i}) max(bi​1​,bi+1​ai​​)≤max(bi+1​1​,bi​ai+1​​)

⇕\Updownarrow ⇕

同乘  bi×bi+1\text{同乘}\;b_i\times b_{i+1} 同乘bi​×bi+1​

max⁡(bi+1  ,  ai×bi)≤max⁡(bi  ,  ai+1×bi+1)\max\bigg(b_{i+1}\;,\;a_i\times b_i \bigg)\le\max\bigg(b_i\;,\;a_{i+1}\times b_{i+1}\bigg) max(bi+1​,ai​×bi​)≤max(bi​,ai+1​×bi+1​)

而由于均为正整数，易知有

bi+1≤ai+1×bi+1    ,    ai×bi≥bib_{i+1}\le a_{i+1}\times b_{i+1}\;\;,\;\;a_i\times b_i\ge b_i bi+1​≤ai+1​×bi+1​,ai​×bi​≥bi​

所以

max⁡(bi+1  ,  ai×bi)≤max⁡(bi  ,  ai+1×bi+1)  ⟺  ai×bi≤ai+1×bi+1\max\bigg(b_{i+1}\;,\;a_i\times b_i \bigg)\le\max\bigg(b_i\;,\;a_{i+1}\times b_{i+1}\bigg)\iff a_i\times b_i \le a_{i+1}\times b_{i+1} max(bi+1​,ai​×bi​)≤max(bi​,ai+1​×bi+1​)⟺ai​×bi​≤ai+1​×bi+1​

进而

s1≤s2  ⟺  ai×bi≤ai+1×bi+1s_1\le s_2\iff a_i\times b_i \le a_{i+1}\times b_{i+1} s1​≤s2​⟺ai​×bi​≤ai+1​×bi+1​

所以相邻位置满足结论。

而所有相邻位置都满足上式时(即不为逆序对)，答案取到最优解。

显然是可以满足所有相邻位置不为逆序对的。即结论所述正确。

故可将原序列排序后模拟计算答案。

注意这题需要 毒瘤高精， 具体为

  • 高精乘

  • 高精除以低精

  • 高精比大小

详见代码。

/**********************************

Problem: [NOIpTG2012] - 国王游戏 

State: Accepted

From: [NOIp2012]

Algorithm: 贪心, (毒瘤)高精 

Author: cyh_toby

Last updated on 2020.3.17

**********************************/
#include <cstdio>
#include <cstring>
#include <algorithm> 
using namespace std;
const int N = 1005, M = 4005;
int n;
int a[N], b[N], id[N];
int c[M], d[M], z[M], res[M], h[M], ans[M];

inline bool cmp(int x, int y) {
	return a[x]*b[x] < a[y]*b[y];
}

inline void mult(int x[], int y[], int r[]) {
	for (int i = 0; i <= x[0]; i++)
		c[i] = x[i];
	for (int i = 0; i <= y[0]; i++)
		d[i] = y[i];
	memset(z, 0 ,sizeof(z));
	for (int i = 1; i <= c[0]; i++) {
		for (int j = 1; j <= d[0]; j++) {
			z[i+j-1] += c[i]*d[j];
			z[i+j] += z[i+j-1]/10;
			z[i+j-1] %= 10;
		}
	}
	z[0] = c[0] + d[0];
	while (z[z[0]] == 0 && z[0] > 1) z[0]--;
	for (int i = 0; i <= z[0]; i++)
		r[i] = z[i];
	return;
}

inline void divi(int x[], int y, int r[]) {
	int t = 0;
	memset(z, 0, sizeof(z));
	for (int i = x[0]; i > 0; i--)
		z[i] = (t*10+x[i]) / y, t = (t*10+x[i]) % y;
	z[0] = x[0];
	while (z[z[0]] == 0 && z[0] > 1) z[0]--;
	for (int i = 0; i <= z[0]; i++)
		r[i] = z[i];
	return;
}

inline bool mx(int x[], int y[]) {
	if (x[0] > y[0]) return 1;
	if (x[0] < y[0]) return 0;
	for (int i = x[0]; i > 0; i--) {
		if (x[i]==y[i]) continue;
		if (x[i] > y[i]) return 1;
		else return 0;
	}
	return 1;
}

inline int* cg(int x) {
	memset(z, 0, sizeof(z));
	while (x) {
		z[++z[0]] = x%10;
		x /= 10;
	}
	if (z[0] == 0) z[0]++;
	return z;
} 

inline void wt(int x[]) {
	for (int i = x[0]; i > 0; i--) 
		putchar(x[i]+'0');
	return;
}

int main()
{
	scanf("%d", &n);
	int t;
	scanf("%d", &t);
	while (t)
		res[++res[0]] = t % 10, t /= 10;
	if (res[0]==0) res[0]++;
	scanf("%d", &t);
	for (int i = 1; i <= n; i++) {
		scanf("%d%d", &a[i], &b[i]);
		id[i] = i;
	}
	sort(id+1, id+1+n, cmp);
	ans[0] = 1;
	for (int i = 1; i <= n; i++) {
		divi(res, b[id[i]], h);
		if (mx(h, ans))
			for (int j = 0; j <= h[0]; j++)
				ans[j] = h[j];
		mult(res, cg(a[id[i]]), res);
	}
	wt(ans);
	return 0;
}
