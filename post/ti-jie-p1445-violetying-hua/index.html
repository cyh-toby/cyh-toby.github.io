原题：P1445 [Violet]樱花

考虑变形为函数以简化问题。

y=(x−n!+n!)×n!x−n!=n!+(n!)2x−n!y=\dfrac{\big(x-n!+n!)\times n!}{x-n!} = n!+\dfrac{(n!)^2}{x-n!} y=x−n!(x−n!+n!)×n!​=n!+x−n!(n!)2​

#include <cstdio>
using namespace std;
#define ll long long
const int N = 1000005, mod = 1000000007;
int n, prime[N], cnt, book[N];
ll ans = 1ll;
inline void init() {
	for (int i = 2; i <= n; i++) {
		if (!book[i])
			prime[++cnt] = i;
		for (int j = 1; prime[j] <= n/i; j++) {
			book[prime[j]*i] = 1;
			if (i % prime[j]==0)
				break;
		}
	}
	return;
}
int main()
{
	scanf("%d", &n);
	init(); 
	for (int i = 1; i <= cnt; i++) {
		int p = prime[i];
		int res = 0;
		for (ll j = p; j <= n;j *= p) {
			res += n/j;
		}
		ans = ans * (2ll*res+1ll) % mod;
	}
	printf("%lld", ans);
	return 0;
}
